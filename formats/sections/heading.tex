% ----- information in a heading ----------------------------------------------

\pgfkeys{
  /personalData/.is family,
  /personalData,
  name/.store in = \name,
  phone/.store in = \phone,
  email/.store in = \email,
  location/.store in = \location
}

% ----- command to add a heading ----------------------------------------------

% Example usage:
%
% \personalData[%
%   name={Rezha Adrian Tanuharja},
%   phone={(+1)12 3456 7890},
%   email={rezha.tanuharja@outlook.com},
%   location={Houston -- U.S.}
% ]

\newcommand{\personalData}[1][]{%
  \pgfkeys{/personalData, #1}%
  \begin{center}
    \Huge{\textbf{\name}}\\
    \vspace{0.25\nominalGap}
    \large{\href{mailto:\email}{\email}}
    \hspace{0.25\nominalGap}\textbullet\hspace{0.25\nominalGap}
    \large{\phone}
    \hspace{0.25\nominalGap}\textbullet\hspace{0.25\nominalGap}
    \large{\location}
  \end{center}
  \par
}
